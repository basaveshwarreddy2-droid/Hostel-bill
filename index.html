async function checkFee() { const rollInput = document.getElementById('rollNumber').value.trim().toUpperCase(); const resultDiv = document.getElementById('result'); // Replace the URL below with YOUR SheetDB API URL const apiUrl = "https://sheetdb.io/api/v1/YOUR_API_ID_HERE"; try { const response = await fetch(apiUrl); const students = await response.json(); const student = students.find(s => s.RollNumber.toUpperCase() === rollInput); if (student) { resultDiv.style.display = "block"; resultDiv.innerHTML = `
Name: ${student.Name}
Roll No: ${student.RollNumber}
Hostel Mess Due: ₹${student.FeeDue}
Status: ${student.Status}
`; } else { resultDiv.style.display = "block"; resultDiv.innerHTML = "No record found. Check the Roll Number."; } } catch (error) { console.error("Error fetching data:", error); resultDiv.innerHTML = "Error loading data. Try again later."; } }
