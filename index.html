import pytesseract from PIL import Image import pandas as pd import os # 1. Configuration - Set path to your Tesseract executable if on Windows # pytesseract.pytesseract.tesseract_cmd = r'C:\Program Files\Tesseract-OCR\tesseract.exe' def extract_data_from_image(image_path): # Load image img = Image.open(image_path) # Extract text from image raw_text = pytesseract.image_to_string(img) # Simple logic to split lines and columns lines = raw_text.split('\n') data = [] for line in lines: parts = line.split() if len(parts) >= 3: # Ensuring we have Roll No, Name, and Dues data.append(parts) return data # 2. Process all images in a folder folder_path = './mess_images' # Put your .jpg files here all_records = [] for filename in os.listdir(folder_path): if filename.endswith(".jpg"): file_path = os.path.join(folder_path, filename) print(f"Processing {filename}...") all_records.extend(extract_data_from_image(file_path)) # 3. Save to CSV df = pd.DataFrame(all_records) df.to_csv('hostel_dues_2025_2026.csv', index=False) print("Extraction complete! File saved as hostel_dues_2025_2026.csv")
